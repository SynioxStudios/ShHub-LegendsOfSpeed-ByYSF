Tabs.Farm:AddToggle("AutoFarmOrbs", {

Tabs.Farm:AddToggle("AutoHoop", {
    Title = "⬆️ Auto Hoop",
    Default = false,
    Callback = function(state)
        if state then
            hoopConnection = game:GetService("RunService").RenderStepped:Connect(function()
                local root = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                local hoops = game.Workspace:FindFirstChild("Hoops")
                if root and hoops then
                    for _, hoop in ipairs(hoops:GetChildren()) do
                        if hoop.Name == "Hoop" or hoop:FindFirstChild("Apperance") then
                            hoop.CFrame = root.CFrame
                            game:GetService("ReplicatedStorage").rEvents.hoopEvent:FireServer("finishedHoop")
                        end
                    end
                end
            end)
        else
            if hoopConnection then
                hoopConnection:Disconnect()
                hoopConnection = nil
            end
        end
    end
})

Tabs.Farm:AddToggle("AutoRace", {
    Title = "🏁 Auto Race/Win",
    Default = false,
    Callback = function(state)
        autoRace = state
    end
})

task.spawn(function()
    while true do
        task.wait(0.5)
        if autoRace then
            pcall(function()
                local player = game.Players.LocalPlayer
                if player.PlayerGui.gameGui.raceJoinLabel.Visible then
                    game:GetService("ReplicatedStorage").rEvents.raceEvent:FireServer("joinRace")
                end
                local raceMaps = game.Workspace:FindFirstChild("raceMaps")
                if raceMaps and player.currentMap.Value ~= "" then
                    for _, map in ipairs(raceMaps:GetChildren()) do
                        if player.currentMap.Value == map.mapValue.Value then
                            repeat task.wait() until player.PlayerGui.gameGui.countdownLabels.goLabel.Visible or not autoRace
                            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                                player.Character.HumanoidRootPart.CFrame = map.finishPart.CFrame
                            end
                        end
                    end
                end
            end)
        end
    end
end)

Tabs.Misc:AddToggle("NoClip", {
    Title = "👻 No Clip",
    Default = false,
    Callback = function(bool)
        noclipEnabled = bool
        if noclipEnabled then
            if noclipConnection then noclipConnection:Disconnect() end
            noclipConnection = game:GetService("RunService").Stepped:Connect(function()
                if noclipEnabled and game.Players.LocalPlayer.Character then
                    for _, v in pairs(game.Players.LocalPlayer.Character:GetDescendants()) do
                        if v:IsA("BasePart") then
                            v.CanCollide = false
                        end
                    end
                end
            end)
        else
            if noclipConnection then
                noclipConnection:Disconnect()
                noclipConnection = nil
            end
            if game.Players.LocalPlayer.Character then
                for _, part in pairs(game.Players.LocalPlayer.Character:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.CanCollide = true
                    end
                end
            end
        end
    end
})
