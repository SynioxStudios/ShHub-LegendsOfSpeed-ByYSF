local SynioxLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/ByYSF-FLuent-U--V4/refs/heads/main/SynioxUi.txt"))()
local displayName = game.Players.LocalPlayer.DisplayName

SynioxLib:StartFlowRGB(
    Color3.fromRGB(120, 0, 0),
    Color3.fromRGB(40, 0, 0),
    Color3.fromRGB(0, 0, 0)
)

local Window = SynioxLib:CreateWindow({
    Title = "Syniox Hub | LoS || HELLO - " .. displayName,
    SubTitle = "v4",
    TabWidth = 160,
    Size = UDim2.fromOffset(450, 400),
    Acrylic = true,
    Theme = "SpeedHubX"
})

local MainFarm = Window:AddTab({ Title = "Main", Icon = "home" })
local RebirthTab = Window:AddTab({ Title = "Rebirths", Icon = "refresh-cw" })
local PetsTab = Window:AddTab({ Title = "Pets & Aura", Icon = "dog" })
local Misc = Window:AddTab({ Title = "Misc", Icon = "settings" })

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")
local player = game.Players.LocalPlayer

local selectedOrb = ""
local orbDropdown = MainFarm:AddDropdown("OrbDropdown", {
    Title = "💎 Select Orb",
    Values = {"Yellow Orb", "Red Orb", "Blue Orb", "Orange Orb", "Ethereal Orb", "Gem"},
    Default = "",
    Callback = function(Value)
        selectedOrb = Value
    end
})

MainFarm:AddToggle("AutoOrb", {
    Title = "🌕 Auto Farm Orbs",
    Default = false,
    Callback = function(state)
        _G.farming = state
        if _G.farming then
            task.spawn(function()
                while _G.farming do
                    if selectedOrb ~= "" then
                        ReplicatedStorage.rEvents.orbEvent:FireServer("collectOrb", selectedOrb, "City")
                    end
                    task.wait(0.001)
                end
            end)
        end
    end
})

local hoopConnection
MainFarm:AddToggle("AutoHoop", {
    Title = "⬆️ Auto Hoop",
    Default = false,
    Callback = function(state)
        if state then
            hoopConnection = RunService.RenderStepped:Connect(function()
                local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                local hoops = game.Workspace:FindFirstChild("Hoops")
                if root and hoops then
                    for _, hoop in ipairs(hoops:GetChildren()) do
                        if hoop.Name == "Hoop" or hoop:FindFirstChild("Apperance") then
                            hoop.CFrame = root.CFrame
                            ReplicatedStorage.rEvents.hoopEvent:FireServer("finishedHoop")
                        end
                    end
                end
            end)
        else
            if hoopConnection then hoopConnection:Disconnect() hoopConnection = nil end
        end
    end
})

MainFarm:AddToggle("AutoRace", {
    Title = "🏁 Auto Race/ Win",
    Default = false,
    Callback = function(state)
        _G.autoRace = state
        if _G.autoRace then
            task.spawn(function()
                while _G.autoRace do
                    task.wait(0.5)
                    pcall(function()
                        if player.PlayerGui.gameGui.raceJoinLabel.Visible then
                            ReplicatedStorage.rEvents.raceEvent:FireServer("joinRace")
                        end
                        local raceMaps = game.Workspace:FindFirstChild("raceMaps")
                        if raceMaps and player.currentMap.Value ~= "" then
                            for _, map in ipairs(raceMaps:GetChildren()) do
                                if player.currentMap.Value == map.mapValue.Value then
                                    repeat task.wait() until player.PlayerGui.gameGui.countdownLabels.goLabel.Visible or not _G.autoRace
                                    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                                        player.Character.HumanoidRootPart.CFrame = map.finishPart.CFrame
                                    end
                                end
                            end
                        end
                    end)
                end
            end)
        end
    end
})

local targetRebirthValue = 0
local infiniteToggle
local targetToggle

RebirthTab:AddInput("RebirthTarget", {
    Title = "🔢 Rebirth Target",
    Default = "",
    Numeric = true,
    Finished = true,
    Callback = function(text)
        targetRebirthValue = tonumber(text) or 0
    end
})

targetToggle = RebirthTab:AddToggle("TargetRebirth", {
    Title = "🔄 Auto Rebirth Target",
    Default = false,
    Callback = function(bool)
        _G.targetRebirthActive = bool
        if bool then
            if _G.infiniteRebirthActive then infiniteToggle:SetValue(false) end
            task.spawn(function()
                while _G.targetRebirthActive do
                    task.wait(0.1)
                    if player.leaderstats.Rebirths.Value >= targetRebirthValue then
                        targetToggle:SetValue(false)
                        break
                    end
                    ReplicatedStorage.rEvents.rebirthEvent:FireServer("rebirthRequest")
                end
            end)
        end
    end
})

infiniteToggle = RebirthTab:AddToggle("InfiniteRebirth", {
    Title = "♾️ Auto Rebirth (Infinitely)",
    Default = false,
    Callback = function(bool)
        _G.infiniteRebirthActive = bool
        if bool then
            if _G.targetRebirthActive then targetToggle:SetValue(false) end
            task.spawn(function()
                while _G.infiniteRebirthActive do
                    task.wait(0.1)
                    ReplicatedStorage.rEvents.rebirthEvent:FireServer("rebirthRequest")
                end
            end)
        end
    end
})

PetsTab:AddSection("💰 Auto Buy")
local selectedCrystal = "Blue Crystal"
PetsTab:AddDropdown("CrystalDropdown", {
    Title = "💎 Select Crystal",
    Values = {"Blue Crystal", "Red Crystal", "Lightning Crystal", "Inferno Crystal", "Lava Crystal", "Snow Crystal", "Electro Crystal", "Desert Crystal"},
    Default = "Blue Crystal",
    Callback = function(Value) selectedCrystal = Value end
})

PetsTab:AddToggle("AutoHatchCryToggle", {
    Title = "🥚 Auto Open Crystal",
    Default = false,
    Callback = function(bool)
        _G.AutoHatchCry = bool
        if bool then
            task.spawn(function()
                while _G.AutoHatchCry do
                    local cry = ReplicatedStorage.openCrystalsFolder:FindFirstChild(selectedCrystal)
                    if cry then ReplicatedStorage.rEvents.openCrystalRemote:InvokeServer("openCrystal", cry.Name) end
                    task.wait(0.1)
                end
            end)
        end
    end
})

Misc:AddToggle("AntiAfkToggle", {
    Title = "⏳ Anti Afk",
    Default = false,
    Callback = function(Value)
        _G.AntiAfkActive = Value
        if Value then
            local vu = game:GetService("VirtualUser")
            antiAfkConnection = player.Idled:Connect(function()
                if _G.AntiAfkActive then vu:CaptureController() vu:ClickButton2(Vector2.new()) end
            end)
        else
            if antiAfkConnection then antiAfkConnection:Disconnect() end
        end
    end
})

Misc:AddToggle("LockPositionToggle", {
    Title = "🔒 Lock Position",
    Default = false,
    Callback = function(Value)
        if Value then
            local currentPos = player.Character.HumanoidRootPart.CFrame
            getgenv().posLock = RunService.Heartbeat:Connect(function()
                if player.Character:FindFirstChild("HumanoidRootPart") then player.Character.HumanoidRootPart.CFrame = currentPos end
            end)
        else
            if getgenv().posLock then getgenv().posLock:Disconnect() getgenv().posLock = nil end
        end
    end
})

Misc:AddToggle("AutoFortuneWheelToggle", {
    Title = "🎰 Auto Fortune Wheel",
    Default = false,
    Callback = function(Value)
        _G.autoFortuneWheelActive = Value
        if Value then
            task.spawn(function()
                while _G.autoFortuneWheelActive do
                    ReplicatedStorage.rEvents.openFortuneWheelRemote:InvokeServer("openFortuneWheel", ReplicatedStorage.fortuneWheelChances["Fortune Wheel"])
                    task.wait()
                end
            end)
        end
    end
})

Misc:AddToggle("NoClipToggle", {
    Title = "👻 No Clip",
    Default = false,
    Callback = function(bool)
        _G.noclipEnabled = bool
        if bool then
            RunService.Stepped:Connect(function()
                if _G.noclipEnabled and player.Character then
                    for _, part in pairs(player.Character:GetDescendants()) do
                        if part:IsA("BasePart") then part.CanCollide = false end
                    end
                end
            end)
        end
    end
})
