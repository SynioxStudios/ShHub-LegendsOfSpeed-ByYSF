local MainFarm = Window:AddTab({ Title = "Main", Icon = "home" })

local selectedOrb = ""

MainFarm:AddDropdown("SelectMap", {
    Title = "📍 Select Map",
    Values = {"City", "Snow City", "Magma City", "Speed Jungle", "Legends Highway"},
    Default = "City",

MainFarm:AddDropdown("SelectOrb", {
    Title = "💎 Select Orb",
    Values = {"Yellow Orb", "Red Orb", "Blue Orb", "Orange Orb", "Ethereal Orb", "Gem"},
    Default = "Yellow Orb",
    Callback = function(Value)
        selectedOrb = Value
    end
})

MainFarm:AddToggle("AutoFarmOrbs", {
    Title = "Auto Farm Orbs",
    Default = false,
    Callback = function(state)
        farming = state
        if farming then
            task.spawn(function()
                while farming do
                    if selectedOrb ~= "" and selectedMap ~= "" then
                        ReplicatedStorage.rEvents.orbEvent:FireServer("collectOrb", selectedOrb, selectedMap)
                    end
                    task.wait(0.001)
                end
            end)
        end
    end
})

MainFarm:AddToggle("AutoHoop", {
    Title = "⬆️ Auto Hoop",
    Default = false,
    Callback = function(state)
        if state then
            hoopConnection = RunService.RenderStepped:Connect(function()
                local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                local hoops = game.Workspace:FindFirstChild("Hoops")
                if root and hoops then
                    for _, hoop in ipairs(hoops:GetChildren()) do
                        if hoop.Name == "Hoop" or hoop:FindFirstChild("Apperance") then
                            hoop.CFrame = root.CFrame
                            ReplicatedStorage.rEvents.hoopEvent:FireServer("finishedHoop")
                        end
                    end
                end
            end)
        else
            if hoopConnection then
                hoopConnection:Disconnect()
                hoopConnection = nil
            end
        end
    end
})

MainFarm:AddToggle("AutoRace", {
    Title = "🏁 Auto Race/Win",
    Default = false,
    Callback = function(state)
        autoRace = state
    end
})

task.spawn(function()
    while true do
        task.wait(0.5)
        if autoRace then
            pcall(function()
                if player.PlayerGui.gameGui.raceJoinLabel.Visible then
                    ReplicatedStorage.rEvents.raceEvent:FireServer("joinRace")
                end
                local raceMaps = game.Workspace:FindFirstChild("raceMaps")
                if raceMaps and player.currentMap.Value ~= "" then
                    for _, map in ipairs(raceMaps:GetChildren()) do
                        if player.currentMap.Value == map.mapValue.Value then
                            repeat task.wait() until player.PlayerGui.gameGui.countdownLabels.goLabel.Visible or not autoRace
                            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                                player.Character.HumanoidRootPart.CFrame = map.finishPart.CFrame
                            end
                        end
                    end
                end
            end)
        end
    end
end)

local RebirthTab = Window:AddTab({ Title = "Rebirths", Icon = "refresh-cw" })
local targetRebirthValue = 0

RebirthTab:AddInput("RebirthTarget", {
    Title = "Rebirth Target",
    Default = "0",
    Numeric = true,
    Finished = true,
    Callback = function(Value)
        targetRebirthValue = tonumber(Value) or 0
    end
})

RebirthTab:AddToggle("AutoRebirthTarget", {
    Title = "🔄 Auto Rebirth Target",
    Default = false,
    Callback = function(bool)
        _G.targetRebirthActive = bool
        if bool then
            task.spawn(function()
                while _G.targetRebirthActive do
                    local currentRebirths = game.Players.LocalPlayer.leaderstats.Rebirths.Value
                    if currentRebirths >= targetRebirthValue then
                        _G.targetRebirthActive = false
                        break
                    end
                    game:GetService("ReplicatedStorage").rEvents.rebirthEvent:FireServer("rebirthRequest")
                    task.wait(0.1)
                end
            end)
        end
    end
})

local Misc = Window:AddTab({ Title = "Misc", Icon = "settings" })

Misc:AddToggle("AntiAfk", {
    Title = "⏳ Anti Afk",
    Default = false,
    Callback = function(Value)
        _G.AntiAfkActive = Value
        if Value then
            local vu = game:GetService("VirtualUser")
            antiAfkConn = game.Players.LocalPlayer.Idled:Connect(function()
                if _G.AntiAfkActive then 
                    vu:CaptureController() 
                    vu:ClickButton2(Vector2.new()) 
                end
            end)
            local ScreenGui = Instance.new("ScreenGui", game:GetService("CoreGui"))
            ScreenGui.Name = "Syniox_AntiAFK"
            local Frame = Instance.new("Frame", ScreenGui)
            Frame.BackgroundColor3 = Color3.new(0,0,0)
            Frame.Size = UDim2.new(0, 120, 0, 125)
            Frame.Position = UDim2.new(0.05, 0, 0.4, 0)
            local tLabel = Instance.new("TextLabel", Frame)
            tLabel.Size = UDim2.new(1,0,1,0)
            tLabel.TextColor3 = Color3.new(1,1,1)
            tLabel.BackgroundTransparency = 1
            tLabel.Font = Enum.Font.SourceSansBold
            local startTime = os.time()
            task.spawn(function()
                while _G.AntiAfkActive do
                    local diff = os.time() - startTime
                    local fps = math.floor(1/game:GetService("RunService").RenderStepped:Wait())
                    tLabel.Text = string.format("Syniox Anti-Afk\n\n⏳ %02d:%02d:%02d\n🚀 FPS: %d", math.floor(diff/3600), math.floor((diff%3600)/60), diff%60, fps)
                    task.wait(1)
                end
            end)
        else
            if antiAfkConn then antiAfkConn:Disconnect() end
            if game:GetService("CoreGui"):FindFirstChild("Syniox_AntiAFK") then 
                game:GetService("CoreGui").Syniox_AntiAFK:Destroy() 
            end
        end
    end
})

    Callback = function(bool)
        _G.noclipEnabled = bool
        game:GetService("RunService").Stepped:Connect(function()
            if _G.noclipEnabled and game.Players.LocalPlayer.Character then
