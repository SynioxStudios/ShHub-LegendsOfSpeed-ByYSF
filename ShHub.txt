local SynioxLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/ByYSF-FLuent-U--V4/refs/heads/main/SynioxUi.txt"))()
local displayName = game.Players.LocalPlayer.DisplayName

SynioxLib:StartFlowRGB(
    Color3.fromRGB(120, 0, 0),
    Color3.fromRGB(40, 0, 0),
    Color3.fromRGB(0, 0, 0)
)

local Window = SynioxLib:CreateWindow({
    Title = "Syniox Hub | LoS || HELLO - " .. displayName,
    SubTitle = "v4",
    TabWidth = 160,
    Size = UDim2.fromOffset(450, 480),
    Acrylic = true,
    Theme = "SpeedHubX"
})

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")
local player = game.Players.LocalPlayer

local MainFarm = Window:AddTab({ Title = "Main", Icon = "home" })

local selectedOrb = ""
local selectedMap = ""
local farming = false
local autoRace = false
local hoopConnection

MainFarm:AddDropdown("SelectMap", {
    Title = "📍 Select Map",
    Values = {"City", "Snow City", "Magma City", "Speed Jungle", "Legends Highway"},
    Default = "City",
    Callback = function(Value)
        selectedMap = Value
    end
})

MainFarm:AddDropdown("SelectOrb", {
    Title = "💎 Select Orb",
    Values = {"Yellow Orb", "Red Orb", "Blue Orb", "Orange Orb", "Ethereal Orb", "Gem"},
    Default = "Yellow Orb",
    Callback = function(Value)
        selectedOrb = Value
    end
})

MainFarm:AddToggle("AutoFarmOrbs", {
    Title = "Auto Farm Orbs",
    Default = false,
    Callback = function(state)
        farming = state
        if farming then
            task.spawn(function()
                while farming do
                    if selectedOrb ~= "" and selectedMap ~= "" then
                        ReplicatedStorage.rEvents.orbEvent:FireServer("collectOrb", selectedOrb, selectedMap)
                    end
                    task.wait(0.001)
                end
            end)
        end
    end
})

MainFarm:AddToggle("AutoHoop", {
    Title = "⬆️ Auto Hoop",
    Default = false,
    Callback = function(state)
        if state then
            hoopConnection = RunService.RenderStepped:Connect(function()
                local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                local hoops = game.Workspace:FindFirstChild("Hoops")
                if root and hoops then
                    for _, hoop in ipairs(hoops:GetChildren()) do
                        if hoop.Name == "Hoop" or hoop:FindFirstChild("Apperance") then
                            hoop.CFrame = root.CFrame
                            ReplicatedStorage.rEvents.hoopEvent:FireServer("finishedHoop")
                        end
                    end
                end
            end)
        else
            if hoopConnection then
                hoopConnection:Disconnect()
                hoopConnection = nil
            end
        end
    end
})

MainFarm:AddToggle("AutoRace", {
    Title = "🏁 Auto Race/Win",
    Default = false,
    Callback = function(state)
        autoRace = state
    end
})

task.spawn(function()
    while true do
        task.wait(0.5)
        if autoRace then
            pcall(function()
                if player.PlayerGui.gameGui.raceJoinLabel.Visible then
                    ReplicatedStorage.rEvents.raceEvent:FireServer("joinRace")
                end
                local raceMaps = game.Workspace:FindFirstChild("raceMaps")
                if raceMaps and player.currentMap.Value ~= "" then
                    for _, map in ipairs(raceMaps:GetChildren()) do
                        if player.currentMap.Value == map.mapValue.Value then
                            repeat task.wait() until player.PlayerGui.gameGui.countdownLabels.goLabel.Visible or not autoRace
                            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                                player.Character.HumanoidRootPart.CFrame = map.finishPart.CFrame
                            end
                        end
                    end
                end
            end)
        end
    end
end)
