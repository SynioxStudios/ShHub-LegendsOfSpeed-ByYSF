local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/ByYusufGui-Syniox/refs/heads/main/SynioxGui.txt", true))()
local displayName = game.Players.LocalPlayer.DisplayName

local window = library:AddWindow("Syniox Hub | Legends Of Speed || HELLO - ".. displayName, {
    title_bar = {
        Color3.fromRGB(160, 0, 0), 
        Color3.fromRGB(40, 0, 0),  
        Color3.fromRGB(0, 0, 0)    
    }, 
    title_bar_transparency = 0.1, 
    background = {
        Color3.fromRGB(0, 0, 0), 
        Color3.fromRGB(10, 10, 10), 
        Color3.fromRGB(20, 20, 20)
    }, 
    background_transparency = 0.1, 
    main_color = Color3.fromRGB(160, 0, 0), 
    min_size = Vector2.new(450, 450), 
    can_resize = true 
})

local farm = window:AddTab("Farm")
farm:AddLabel("Fast Farm")

local selectedOrb = ""
local farming = false

local orbDropdown = farm:AddDropdown("Select Orb", function(text)
    selectedOrb = text
end)

local orbList = {
    "Yellow Orb", "Red Orb", "Blue Orb", "Orange Orb", "Ethereal Orb"
}

for _, name in ipairs(orbList) do 
    orbDropdown:Add(name) 
end

farm:AddSwitch("Auto Farm Orbs", function(state)
    farming = state
    if farming then
        task.spawn(function()
            while farming do
                if selectedOrb ~= "" then
                    game:GetService("ReplicatedStorage").rEvents.orbEvent:FireServer("collectOrb", selectedOrb, "City")
                end
                task.wait(0.001)
            end
        end)
    end
end)

local hoopConnection
local fakeMotion

farm:AddSwitch("⬆️ Fast Auto Hoop ", function(state)
    local hooping = state
    
    if hooping then
        fakeMotion = game:GetService("RunService").Heartbeat:Connect(function()
            local root = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if root then
                root.Velocity = Vector3.new(0, 0.01, 0)
                root.AssemblyLinearVelocity = Vector3.new(50, 0, 50)
            end
        end)

        hoopConnection = game:GetService("RunService").RenderStepped:Connect(function()
            local root = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            local hoops = game.Workspace:FindFirstChild("Hoops")
            
            if root and hoops then
                for _, hoop in ipairs(hoops:GetChildren()) do
                    if hoop.Name == "Hoop" or hoop:FindFirstChild("Apperance") then
                        hoop.CFrame = root.CFrame
                        game:GetService("ReplicatedStorage").rEvents.hoopEvent:FireServer("finishedHoop")
                    end
                end
            end
        end)
    else
        if hoopConnection then hoopConnection:Disconnect() hoopConnection = nil end
        if fakeMotion then fakeMotion:Disconnect() fakeMotion = nil end
        
        local root = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if root then root.Velocity = Vector3.new(0,0,0) end
    end
end)
