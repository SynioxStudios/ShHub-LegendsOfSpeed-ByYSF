local MainFarm = Window:AddTab({ Title = "Main", Icon = "home" })
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local Players = game:GetService("Players")
local player = Players.LocalPlayer

local selectedOrb = ""
local orbDropdown = MainFarm:AddDropdown("OrbDropdown", {
    Title = "💎 Select Orb",
    Values = {},
    Default = "",
    Callback = function(Value)
        selectedOrb = Value
    end
})

local orbList = {"Yellow Orb", "Red Orb", "Blue Orb", "Orange Orb", "Ethereal Orb", "Gem"}
for _, name in ipairs(orbList) do orbDropdown:Add(name) end

MainFarm:AddToggle("AutoOrb", {
    Title = "🌕 Auto Farm Orbs",
    Default = false,
    Callback = function(state)
        _G.farming = state
        if _G.farming then
            task.spawn(function()
                while _G.farming do
                    if selectedOrb ~= "" then
                        ReplicatedStorage.rEvents.orbEvent:FireServer("collectOrb", selectedOrb, "City")
                    end
                    task.wait(0.001)
                end
            end)
        end
    end
})

local hoopConnection
MainFarm:AddToggle("AutoHoop", {
    Title = "⬆️ Auto Hoop",
    Default = false,
    Callback = function(state)
        if state then
            hoopConnection = RunService.RenderStepped:Connect(function()
                local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                local hoops = game.Workspace:FindFirstChild("Hoops")
                if root and hoops then
                    for _, hoop in ipairs(hoops:GetChildren()) do
                        if hoop.Name == "Hoop" or hoop:FindFirstChild("Apperance") then
                            hoop.CFrame = root.CFrame
                            ReplicatedStorage.rEvents.hoopEvent:FireServer("finishedHoop")
                        end
                    end
                end
            end)
        else
            if hoopConnection then hoopConnection:Disconnect() hoopConnection = nil end
        end
    end
})

MainFarm:AddToggle("AutoRace", {
    Title = "🏁 Auto Race/ Win",
    Default = false,
    Callback = function(state)
        _G.autoRace = state
        if _G.autoRace then
            task.spawn(function()
                while _G.autoRace do
                    task.wait(0.5)
                    pcall(function()
                        if player.PlayerGui.gameGui.raceJoinLabel.Visible then
                            ReplicatedStorage.rEvents.raceEvent:FireServer("joinRace")
                        end

                        local raceMaps = game.Workspace:FindFirstChild("raceMaps")
                        if raceMaps and player.currentMap.Value ~= "" then
                            for _, map in ipairs(raceMaps:GetChildren()) do
                                if player.currentMap.Value == map.mapValue.Value then
                                    repeat task.wait() until player.PlayerGui.gameGui.countdownLabels.goLabel.Visible or not _G.autoRace
                                    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                                        player.Character.HumanoidRootPart.CFrame = map.finishPart.CFrame
                                    end
                                end
                            end
                        end
                    end)
                end
            end)
        end
    end
})

local RebirthTab = Window:AddTab({ Title = "Rebirths", Icon = "refresh-cw" })
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local player = game.Players.LocalPlayer

local targetRebirthValue = 0
local infiniteToggle
local targetToggle

RebirthTab:AddInput("RebirthTarget", {
    Title = "🔢 Rebirth Target",
    Default = "",
    Numeric = true,
    Finished = true,
    Callback = function(text)
        targetRebirthValue = tonumber(text) or 0
    end
})

targetToggle = RebirthTab:AddToggle("TargetRebirth", {
    Title = "🔄 Auto Rebirth Target",
    Default = false,
    Callback = function(bool)
        _G.targetRebirthActive = bool
        if bool then
            if _G.infiniteRebirthActive then 
                infiniteToggle:SetValue(false) 
            end
            task.spawn(function()
                while _G.targetRebirthActive do
                    task.wait(0.1)
                    local current = player.leaderstats.Rebirths.Value
                    if current >= targetRebirthValue then
                        targetToggle:SetValue(false)
                        break
                    end
                    ReplicatedStorage.rEvents.rebirthEvent:FireServer("rebirthRequest")
                end
            end)
        end
    end
})

infiniteToggle = RebirthTab:AddToggle("InfiniteRebirth", {
    Title = "♾️ Auto Rebirth (Infinitely)",
    Default = false,
    Callback = function(bool)
        _G.infiniteRebirthActive = bool
        if bool then
            if _G.targetRebirthActive then 
                targetToggle:SetValue(false) 
            end
            task.spawn(function()
                while _G.infiniteRebirthActive do
                    task.wait(0.1)
                    ReplicatedStorage.rEvents.rebirthEvent:FireServer("rebirthRequest")
                end
            end)
        end
    end
})

local Misc = Window:AddTab({ Title = "Misc", Icon = "settings" })

Misc:AddToggle("AntiAfkToggle", {
    Title = "⏳ Anti Afk",
    Default = false,
    Callback = function(Value)
        if Value then
            _G.AntiAfkActive = true
            local vu = game:GetService("VirtualUser")
            antiAfkConnection = game:GetService("Players").LocalPlayer.Idled:Connect(function()
                if _G.AntiAfkActive then
                    vu:CaptureController()
                    vu:ClickButton2(Vector2.new())
                end
            end)

            local ScreenGui = Instance.new("ScreenGui")
            local MainFrame = Instance.new("Frame")
            local UICorner = Instance.new("UICorner")
            local Title = Instance.new("TextLabel")
            local TimerLabel = Instance.new("TextLabel")
            local FPSLabel = Instance.new("TextLabel")
            local MSLabel = Instance.new("TextLabel")

            ScreenGui.Name = "Syniox_AntiAFK"
            ScreenGui.Parent = game:GetService("CoreGui")
            
            MainFrame.Parent = ScreenGui
            MainFrame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
            MainFrame.BackgroundTransparency = 0.1
            MainFrame.Position = UDim2.new(0.05, 0, 0.4, 0)
            MainFrame.Size = UDim2.new(0, 120, 0, 125)
            MainFrame.Active = true
            MainFrame.Draggable = true

            UICorner.CornerRadius = UDim.new(0, 8)
            UICorner.Parent = MainFrame

            Title.Parent = MainFrame
            Title.Size = UDim2.new(1, 0, 0, 28)
            Title.Text = "Syniox Anti-Afk"
            Title.TextColor3 = Color3.fromRGB(200, 0, 0)
            Title.BackgroundTransparency = 1
            Title.TextSize = 15
            Title.Font = Enum.Font.SourceSansBold

            local function SetupLabel(lbl, text, pos)
                lbl.Parent = MainFrame
                lbl.Position = pos
                lbl.Size = UDim2.new(1, 0, 0, 25)
                lbl.Text = text
                lbl.TextColor3 = Color3.fromRGB(255, 255, 255)
                lbl.BackgroundTransparency = 1
                lbl.TextSize = 16
                lbl.Font = Enum.Font.SourceSansBold
                lbl.TextXAlignment = Enum.TextXAlignment.Center
            end

            SetupLabel(TimerLabel, "⏳ Time: 00:00:00", UDim2.new(0, 0, 0, 35))
            SetupLabel(FPSLabel, "🚀 FPS: 0", UDim2.new(0, 0, 0, 62))
            SetupLabel(MSLabel, "📡 MS: 0", UDim2.new(0, 0, 0, 89))

            local startTime = os.time()
            task.spawn(function()
                while _G.AntiAfkActive do
                    local diff = os.time() - startTime
                    TimerLabel.Text = string.format("⏳ Time: %02d:%02d:%02d", math.floor(diff/3600), math.floor((diff%3600)/60), diff%60)
                    
                    local fps = math.floor(1 / game:GetService("RunService").RenderStepped:Wait())
                    FPSLabel.Text = "🚀 FPS: " .. fps
                    
                    local ping = math.floor(game:GetService("Stats").Network.ServerStatsItem["Data Ping"]:GetValue())
                    MSLabel.Text = "📡 MS: " .. ping
                    
                    task.wait(0.5)
                end
            end)
        else
            _G.AntiAfkActive = false
            if antiAfkConnection then antiAfkConnection:Disconnect() end
            local oldGui = game:GetService("CoreGui"):FindFirstChild("Syniox_AntiAFK")
            if oldGui then oldGui:Destroy() end
        end
    end
})
